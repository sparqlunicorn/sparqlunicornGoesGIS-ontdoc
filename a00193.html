31 def detectSubjectType(g,subjectstorender,geoclasslist):
32 subjectsToType={}
33 typeToFields={}

35 typeToFields[str(sub)]=set()
36 for tup in g.predicate_objects(sub):
37 if str(tup[0])=="http://www.w3.org/1999/02/22-rdf-syntax-ns#type" and str(tup[1]) in geoclasslist:
38 subjectsToType[str(sub)]=str(tup[1])
39 typeToFields[str(sub)].add(str(tup[0]))

44 del typeToFields[str(sub)]
